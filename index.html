CREATE TABLE staff (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text NOT NULL,
  pin text NOT NULL,
  leave boolean DEFAULT false,
  salary integer DEFAULT 210000,
  is_hourly boolean DEFAULT false,
  hourly_wage integer DEFAULT 1200,
  default_start text DEFAULT '10:00',
  default_end text DEFAULT '19:00',
  transport integer DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE records (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_id uuid REFERENCES staff(id) ON DELETE CASCADE,
  date date NOT NULL,
  type text CHECK (type IN
    ('cielo','glam','off','summer','newyear','absent')),
  start_time text DEFAULT '10:00',
  end_time text DEFAULT '19:30',
  reason text DEFAULT '',
  time_reason text DEFAULT '',
  created_at timestamptz DEFAULT now(),
  UNIQUE(staff_id, date)
);

CREATE TABLE settings (
  key text PRIMARY KEY,
  value text NOT NULL
);

-- RLSを無効化（社内利用のため）
ALTER TABLE staff DISABLE ROW LEVEL SECURITY;
ALTER TABLE records DISABLE ROW LEVEL SECURITY;
ALTER TABLE settings DISABLE ROW LEVEL SECURITY;
